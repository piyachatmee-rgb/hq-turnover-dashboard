1. แนวทางจัดเตรียมตารางบน BigQuery (Slowly Changing Dimension Concept)

เพื่อตอบรับโจทย์ของพี่หัวหน้าในการเก็บโครงสร้างองค์กรและหัวหน้างาน "ณ วันที่ลาออกจริง" (As-of-Event) และป้องกันข้อมูลเปลี่ยนแปลงเมื่อมีการ Re-org ในอนาคต ให้สร้างตาราง Data View หรือ Data Mart ชื่อ dm_hq_employee_monthly_snapshot ด้วยโครงสร้างดังต่อไปนี้:

CREATE OR REPLACE TABLE `your_project.hr_analytics.dm_hq_employee_monthly_snapshot` AS
WITH calendar AS (
  -- สร้างชุดข้อมูลวันสิ้นเดือนของทุกเดือนย้อนหลัง 3 ปี
  SELECT snapshot_date 
  FROM UNNEST(GENERATE_DATE_ARRAY('2023-01-01', CURRENT_DATE(), INTERVAL 1 MONTH)) AS snapshot_date
),
employee_history AS (
  -- ดึงข้อมูลประวัติพนักงานย้อนหลังรวมถึงสถานะการทำงาน
  SELECT 
    employee_id,
    name,
    job_title,
    job_grade,
    status,
    division,
    department,
    section,
    team,
    manager_id,
    manager_name,
    hire_date,
    termination_date,
    exit_type,
    resignation_reason,
    performance_rating,
    -- ตัวกำหนดช่วงเวลาของ record
    valid_from,
    COALESCE(valid_to, DATE '9999-12-31') AS valid_to
  FROM `your_project.hr_source.employee_history_scd2`
)
SELECT 
  c.snapshot_date AS snapshot_month,
  e.employee_id,
  e.name,
  e.job_title,
  e.job_grade,
  e.division,
  e.department,
  e.section,
  e.team,
  e.manager_id,
  e.manager_name,
  e.hire_date,
  e.termination_date,
  e.exit_type,
  e.resignation_reason,
  e.performance_rating,
  -- ตรวจสอบและตั้งแฟล็กเช็กเพื่อช่วยคำนวณใน Looker Studio
  CASE WHEN e.status = 'Active' AND (e.termination_date IS NULL OR e.termination_date > c.snapshot_date) THEN 1 ELSE 0 END AS is_active,
  CASE WHEN e.status = 'Terminated' AND e.termination_date >= c.snapshot_date AND e.termination_date < DATE_ADD(c.snapshot_date, INTERVAL 1 MONTH) THEN 1 ELSE 0 END AS is_exit
FROM calendar c
JOIN employee_history e 
  ON c.snapshot_date >= e.valid_from AND c.snapshot_date < e.valid_to;

2. สูตรคำนวณอัตราการลาออก (Turnover Rate Formula) ใน Looker Studio

เพื่อหลีกเลี่ยงปัญหา Looker Studio ฟิลเตอร์วันที่แล้วทำให้ประวัติ headcount ต้นงวดหายไป ให้ตั้งค่า Calculated Fields ดังนี้:

  • 1. จำนวนคนลาออกในงวด (Exits in Period)
    สร้างฟิลด์คำนวณใหม่: Exits_Period = SUM(is_exit)
  • 2. จำนวนคนทำงานเฉลี่ยในงวด (Average Headcount in Period)
    สร้างฟิลด์คำนวณใหม่: Avg_Headcount_Period = SUM(is_active) / COUNT_DISTINCT(snapshot_month)
  • 3. อัตราการลาออกร้อยละ (Turnover Rate %)
    สร้างฟิลด์คำนวณใหม่: Turnover_Rate_Percent = SUM(is_exit) / (SUM(is_active) / COUNT_DISTINCT(snapshot_month))
    *หมายเหตุ: เลือก Format Type ของฟิลด์คำนวณนี้เป็น Numeric -> Percent เพื่อให้แสดงผลลัพธ์เป็นร้อยละอย่างสวยงาม

3. รายละเอียดการสร้างรายงานแยกหน้า (Recommended Looker Studio Pages Setup)

เราออกแบบโครงสร้างหน้ากระดาษเป็นสัดส่วน 4 หน้า เพื่อตอบสนองความต้องการของผู้ใช้แต่ละกลุ่มดังนี้:

หน้า (Pages) ผู้ใช้งานหลัก เป้าหมายหลักในการเปิดหน้าจอ
1. ภาพรวมผู้บริหาร (Executive Overview) ผู้บริหารระดับสูง, พี่ชล, หัวหน้า People HQ ดูทิศทางการลาออกรวม เทียบเกณฑ์ตั้งรับความเสี่ยง เพื่อหาแผนงานด่วน
2. เจาะลึกรายโครงสร้าง (Org Traceability) HRBP HQ Team, People Analytics วิเคราะห์หาจุดรั่วไหลระดับโครงสร้าง สำนัก -> ฝ่าย -> แผนก เพื่อวางกลยุทธ์ป้องกัน
3. ความรับผิดชอบรายหัวหน้า (Leadership Accountability) HRBP HQ Team (จำกัดสิทธิ์ความปลอดภัย) สืบค้นหัวหน้างานที่มีปัญหาลาออกซ้ำซาก เพื่อเข้าพูดคุย/สนับสนุนแบบรายบุคคล
4. สาเหตุและการลาออก (Exit Reasons & Demographics) HRBP HQ Team, People Operation วิเคราะห์ความสัมพันธ์ระหว่างเหตุผลการลาออก, อายุงาน และระดับประเมินงานคนเก่ง
